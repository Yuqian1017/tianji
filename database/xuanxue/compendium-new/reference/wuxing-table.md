## 五行生克关系

```javascript
const WUXING = {
  generate: { wood:'fire', fire:'earth', earth:'metal', metal:'water', water:'wood' },
  restrict: { wood:'earth', earth:'water', water:'fire', fire:'metal', metal:'wood' },
  
  // 完整五种关系
  relation: function(me, target) {
    if (me === target) return 'sibling';         // 比和
    if (this.generate[me] === target) return 'i_generate';  // 我生
    if (this.generate[target] === me) return 'generates_me'; // 生我
    if (this.restrict[me] === target) return 'i_restrict';  // 我克
    if (this.restrict[target] === me) return 'restricts_me'; // 克我
  }
};
```
