# 六爻飞伏神完整表

> 纯数据。当卦中缺少所需用神时，去本宫卦（八纯卦）找伏神。

---

## 一、伏神规则

1. 卦中找不到需要的六亲 → 在本宫八纯卦对应爻位找
2. 伏在哪个爻下面 → 该爻为「飞神」，被找出来的为「伏神」
3. 飞伏关系决定伏神能否「出来」发挥作用

### 飞伏关系判断

| 关系 | 效果 | 评估 |
|------|------|------|
| 飞生伏 | 伏神得飞神帮助 | 较易出伏 |
| 伏生飞 | 伏神泄气给飞神 | 难出伏 |
| 飞克伏 | 伏神被压制 | 很难出伏 |
| 伏克飞 | 伏神有反抗力 | 可能出伏但有冲突 |
| 飞伏比和 | 和平共处 | 中性 |

### 出伏条件

- 月建或日辰**生助**伏神
- 月建或日辰**冲**飞神（使飞神让位）
- 伏神逢**值**（伏神地支对应的日或月到来）

/**
 * 查找伏神
 * @param palace 卦所在的宫 ('乾','坤'等)
 * @param targetLiuqin 需要的六亲 ('妻财','官鬼'等)
 * @param currentGua 当前卦各爻的纳甲数据
 * @returns 伏神信息
 */
function findFushen(palace, targetLiuqin, currentGua) {
  const pureGua = PURE_GUA_LIUQIN[palace];
  
  for (const line of pureGua.lines) {
    if (line.liuqin === targetLiuqin) {
      const flyingSpirit = currentGua[line.pos - 1]; // 飞神（压在上面的）
      
      return {
        found: true,
        position: line.pos,
        fushen: line,         // 伏神详情
        feishen: flyingSpirit, // 飞神详情
        relationship: analyzeFeiFu(flyingSpirit.wx, line.wx)
      };
    }
  }
  
  return { found: false };
}

function analyzeFeiFu(feiWuxing, fuWuxing) {
  const rel = wuxingRelation(feiWuxing, fuWuxing);
  const labels = {
    'generate': '飞生伏（较易出伏）',
    'generated_by': '伏生飞（难出伏）',
    'restrict': '飞克伏（很难出伏）',
    'restricted_by': '伏克飞（可能出伏）',
    'same': '飞伏比和（中性）'
  };
  return labels[rel] || '未知';
}
