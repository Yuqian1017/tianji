## 三、节气近似计算算法

> 以下算法可用于快速估算，但精度为±1日。高精度排盘请使用天文历表。

```javascript
/**
 * 寿星万年历节气算法（简化版）
 * 精度：±2小时左右
 */

// 节气基准数据（以2000年为基准的儒略日分数）
const JIEQI_BASE = {
  '小寒': 6.11, '大寒': 20.84, '立春': 4.15, '雨水': 19.04,
  '惊蛰': 6.11, '春分': 21.37, '清明': 5.75, '谷雨': 20.60,
  '立夏': 6.31, '小满': 21.73, '芒种': 6.64, '夏至': 22.05,
  '小暑': 7.81, '大暑': 23.52, '立秋': 8.28, '处暑': 23.85,
  '白露': 8.44, '秋分': 23.82, '寒露': 9.10, '霜降': 24.15,
  '立冬': 8.08, '小雪': 22.83, '大雪': 7.53, '冬至': 22.48
};

// 近似公式
function approxJieqiDate(year, jieqiName) {
  const jieqiOrder = [
    '小寒','大寒','立春','雨水','惊蛰','春分',
    '清明','谷雨','立夏','小满','芒种','夏至',
    '小暑','大暑','立秋','处暑','白露','秋分',
    '寒露','霜降','立冬','小雪','大雪','冬至'
  ];
  
  const idx = jieqiOrder.indexOf(jieqiName);
  const month = Math.floor(idx / 2) + 1; // 1-12
  const base = JIEQI_BASE[jieqiName];
  const Y = year - 2000;
  
  // 简化算法
  let day = base + 0.2422 * Y - Math.floor(Y / 4);
  
  // 部分节气有世纪修正
  if (year >= 2000) {
    // 2000年后的修正较小
  }
  
  return {
    year,
    month: month <= 2 ? month : month, // 小寒大寒在上一年
    day: Math.floor(day),
    approxHour: Math.round((day - Math.floor(day)) * 24)
  };
}
```

## 四、判断出生时间属于哪个月柱

```javascript
/**
 * 根据出生年月日时间，确定八字月柱
 * @param year 公历年
 * @param month 公历月
 * @param day 公历日
 * @param hour 时(0-23)
 * @param minute 分(0-59)
 * @returns { monthIdx, branch } 
 */
function determineMonthPillar(year, month, day, hour, minute) {
  // 需要查精确节气时间表
  // 此处以近似值为示意
  
  const jieqiMonths = [
    { name: '小寒', month: 1, approxDay: 6, branch: '丑' },
    { name: '立春', month: 2, approxDay: 4, branch: '寅' },
    { name: '惊蛰', month: 3, approxDay: 6, branch: '卯' },
    { name: '清明', month: 4, approxDay: 5, branch: '辰' },
    { name: '立夏', month: 5, approxDay: 6, branch: '巳' },
    { name: '芒种', month: 6, approxDay: 6, branch: '午' },
    { name: '小暑', month: 7, approxDay: 7, branch: '未' },
    { name: '立秋', month: 8, approxDay: 8, branch: '申' },
    { name: '白露', month: 9, approxDay: 8, branch: '酉' },
    { name: '寒露', month: 10, approxDay: 8, branch: '戌' },
    { name: '立冬', month: 11, approxDay: 7, branch: '亥' },
    { name: '大雪', month: 12, approxDay: 7, branch: '子' }
  ];
  
  // 从后往前找，找到第一个出生日期已过的节
  for (let i = jieqiMonths.length - 1; i >= 0; i--) {
    const jq = jieqiMonths[i];
    if (month > jq.month || (month === jq.month && day >= jq.approxDay)) {
      return { branch: jq.branch, monthIdx: i };
    }
  }
  
  // 如果都没匹配（1月初，小寒之前），则仍属上年十二月（丑月前的子月末）
  return { branch: '子', monthIdx: 10 };
}
```

## 五、真太阳时校正

```javascript
/**
 * 北京时间 → 真太阳时
 * 中国标准时间基准经度：东经120°
 * 每1度经度差 = 4分钟
 */
function toTrueSolarTime(beijingHour, beijingMinute, longitude) {
  const standardLng = 120;
  const diffMinutes = (longitude - standardLng) * 4;
  
  let totalMinutes = beijingHour * 60 + beijingMinute + diffMinutes;
  
  // 还需加上均时差（Equation of Time），此处省略
  // 均时差随季节变化，最大约±16分钟
  
  const hour = Math.floor(totalMinutes / 60) % 24;
  const minute = totalMinutes % 60;
  
  return { hour, minute };
}
```
